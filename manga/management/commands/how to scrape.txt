Basic Usage
bash# Main scraper (default: 10 manga, 5 chapters each)
python manage.py scrape_manga

# Metadata scraper (default: 100 per page, 10 pages)
python manage.py scrape_metadata
With Options
bash# Scrape 20 manga with 10 chapters each
python manage.py scrape_manga --max-manga 20 --max-chapters 10

# Slower scraping (more reliable)
python manage.py scrape_manga --max-manga 5 --delay-min 3 --delay-max 6

# Get 500 manga metadata quickly
python manage.py scrape_metadata --per-page 100 --max-pages 5

# Start from page 3
python manage.py scrape_manga --start-page 3
